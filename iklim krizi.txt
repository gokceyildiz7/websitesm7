  <script>
    document.addEventListener('DOMContentLoaded', () => {
      // Hamburger menü kontrolü
      const menuBtn = document.getElementById('menu-btn');
      const sidePanel = document.getElementById('side-panel');

      menuBtn.addEventListener('click', () => {
        sidePanel.classList.toggle('open');
        const isOpen = sidePanel.classList.contains('open');
        sidePanel.setAttribute('aria-hidden', (!isOpen).toString());
      });
      menuBtn.addEventListener('keypress', e => {
        if (e.key === 'Enter' || e.key === ' ') {
          e.preventDefault();
          menuBtn.click();
        }
      });

      // Menü butonlarıyla sayfa değiştirme
      const menuButtons = document.querySelectorAll('#menu-list button');
      const sections = document.querySelectorAll('main > section');

      function switchSection(targetId) {
        sections.forEach(sec => {
          if (sec.id === targetId) {
            sec.classList.add('active');
            sec.setAttribute('tabindex', '0');
            sec.focus();
          } else {
            sec.classList.remove('active');
            sec.setAttribute('tabindex', '-1');
          }
        });

        menuButtons.forEach(btn => {
          if (btn.dataset.target === targetId) {
            btn.classList.add('active');
            btn.setAttribute('aria-current', 'page');
          } else {
            btn.classList.remove('active');
            btn.removeAttribute('aria-current');
          }
        });

        // Menü panelini kapat
        sidePanel.classList.remove('open');
        sidePanel.setAttribute('aria-hidden', 'true');
      }

      menuButtons.forEach(btn => {
        btn.addEventListener('click', () => {
          switchSection(btn.dataset.target);
        });
        btn.addEventListener('keypress', e => {
          if (e.key === 'Enter' || e.key === ' ') {
            e.preventDefault();
            btn.click();
          }
        });
      });

      // Karbon ayak izi hesaplayıcı
      const startBtn = document.getElementById('startBtn');
      const questionsForm = document.getElementById('questionsForm');
      const calculateBtn = document.getElementById('calculateBtn');
      const resultDiv = document.getElementById('result');

      startBtn.addEventListener('click', () => {
        questionsForm.style.display = questionsForm.style.display === 'block' ? 'none' : 'block';
        resultDiv.style.display = 'none';
      });

      // Form inputlarını izle, hepsi dolunca hesapla butonu aktif olsun
      const inputs = questionsForm.querySelectorAll('input, select');

      function checkFormValidity() {
        let allFilled = true;
        inputs.forEach(input => {
          if (!input.value || (input.tagName === 'SELECT' && input.selectedIndex === 0)) {
            allFilled = false;
          }
        });
        if (allFilled) {
          calculateBtn.classList.add('enabled');
          calculateBtn.disabled = false;
        } else {
          calculateBtn.classList.remove('enabled');
          calculateBtn.disabled = true;
        }
      }

      inputs.forEach(input => {
        input.addEventListener('input', checkFormValidity);
      });

      // Hesaplama fonksiyonu
      calculateBtn.addEventListener('click', () => {
        if (calculateBtn.disabled) return;

        const carKm = parseFloat(document.getElementById('carKm').value);
        const electricity = parseFloat(document.getElementById('electricity').value);
        const flightHours = parseFloat(document.getElementById('flightHours').value);
        const meatConsumption = parseInt(document.getElementById('meatConsumption').value, 10);
        const waste = parseFloat(document.getElementById('waste').value);
        const publicTransport = parseFloat(document.getElementById('publicTransport').value);

        // Basit karbon ayak izi hesabı (ton cinsinden)
        // Araç: 0.2 kg CO2/km, Elektrik: 0.5 kg CO2/kWh, Uçak: 90 kg CO2/saat,
        // Et tüketimi skala 0-3: 0=0, 1=1.5, 2=3, 3=5 (haftalık porsiyon bazında)
        // Çöp: 0.1 kg CO2/kg, Toplu taşıma kullanımı haftada gün sayısı, 0.05 ton azaltım/gün
        let carEmissions = carKm * 0.2 * 365 / 1000; // ton/yıl
        let electricityEmissions = electricity * 0.5 * 12 / 1000; // ton/yıl
        let flightEmissions = flightHours * 90 / 1000; // ton/yıl
        let meatEmissions = meatConsumption * 0.5 * 52; // kg/yıl
        meatEmissions = meatEmissions / 1000; // ton/yıl
        let wasteEmissions = waste * 0.1 * 12 / 1000; // ton/yıl
        let publicTransportReduction = publicTransport * 0.05; // ton/yıl azaltım

        let totalEmissions = carEmissions + electricityEmissions + flightEmissions + meatEmissions + wasteEmissions - publicTransportReduction;
        totalEmissions = Math.max(0, totalEmissions); // negatif olmasın

        // Sonuç yorumlama
        let message = '';
        let cssClass = '';
        if (totalEmissions < 3) {
          message = `<strong>Harika!</strong> Karbon ayak izin düşük seviyede: ${totalEmissions.toFixed(2)} ton CO₂/yıl. Doğaya katkın büyük.`;
          cssClass = 'good';
        } else if (totalEmissions < 7) {
          message = `<strong>Orta düzeyde.</strong> Karbon ayak izin: ${totalEmissions.toFixed(2)} ton CO₂/yıl. Daha az tüketmeye çalışabilirsin.`;
          cssClass = 'medium';
        } else {
          message = `<strong>Yüksek karbon ayak izi!</strong> ${totalEmissions.toFixed(2)} ton CO₂/yıl. Lütfen tüketim alışkanlıklarını gözden geçir.`;
          cssClass = 'bad';
          message += `<br><img src="https://images.unsplash.com/photo-1506744038136-46273834b3fb?auto=format&fit=crop&w=600&q=80" alt="Kuraklık ve su kıtlığı" />`;
        }

        resultDiv.innerHTML = message;
        resultDiv.className = 'result ' + cssClass;
        resultDiv.style.display = 'block';
        resultDiv.scrollIntoView({behavior: "smooth"});
      });

      // Başlangıçta hesapla butonu pasif olsun
      calculateBtn.disabled = true;

      // İletişim formu gönderme simülasyonu
      const contactForm = document.getElementById('contactForm');
      const thankYou = document.getElementById('thankYou');

      contactForm.addEventListener('submit', e => {
        e.preventDefault();
        contactForm.style.display = 'none';
        thankYou.style.display = 'block';
      });
    });
